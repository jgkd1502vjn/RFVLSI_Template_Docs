	\begin{itemize}
		\item All IEEE transactions/bibtex packages. (the \textbf{ieeetran}, and \textbf{biblatex-ieee} package)
		\item \textbf{Textcomp}: support some symbols. (the \textbf{was} package)
		\item \textbf{amsmath}: support some maths.

\end{enumerate}
